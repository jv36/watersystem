350bool Graph::removeVertex(const std::string &in) {
351 for (auto it = vertexSet.begin(); it != vertexSet.end(); it++) {
352 if ((*it)->getCode() == in) {
353 auto v = *it;
354 v->removeOutgoingEdges();
355 for (auto u : vertexSet) {
356 u->removeEdge(v->getCode());
357 }
358 vertexSet.erase(it);
359 delete v;
360 return true;
361 }
362 }
363 return false;
364}

414bool Graph::addBidirectionalEdge(const std::string &sourc, const std::string &dest, double w) {
415 auto v1 = findVertex(sourc);
416 auto v2 = findVertex(dest);
417 if (v1 == nullptr || v2 == nullptr)
418 return false;
419 auto e1 = v1->addEdge(v2, w);
420 auto e2 = v2->addEdge(v1, w);
421 e1->setReverse(e2);
422 e2->setReverse(e1);
