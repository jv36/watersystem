198void Manager::maxWaterFlow(Graph &graph) {
199 // Edmonds-Karp algorithm
200 // Possible approach: create a super-source and a super-sink
201 // connect the super-source to the reservoirs and the super-sink to the cities
202 // run the algorithm and check the flow from the super-source to the super-sink
203 // the flow from the super-source to the super-sink is the maximum flow
204 // then select a city (using code) and check the flow from the super-source to the city
205
206 Vertex* superSource = new Vertex("supersource", "supersource", 999, "SS", INT_MAX);
207 Vertex* superSink = new Vertex("supersink", "supersink", 888, "TS", INT_MAX);
208
209 graph.addVertex(superSource);
210 graph.addVertex(superSink);
211
212 for(auto v: graph.getVertexSet()){
213 if (!v->getCode().empty()) {
214 if (v->getCode().at(0) == 'R') {
215 graph.addEdge(superSource->getCode(), v->getCode(), v->getMaxDelivery());
216 }
217
218 if (v->getCode().at(0) == 'C') {
219 graph.addEdge(v->getCode(), superSink->getCode(), v->getDemand());
220 }
221 }
222 }
223
224 edmondsKarp(graph, superSource, superSink);
225}

315bool Manager::findAugmentingPaths(Graph &graph, Vertex *s, Vertex *t){
316 for(auto v : graph.getVertexSet()) {
317 v->setVisited(false);
318 }
319
320 s->setVisited(true);
321 std::queue<Vertex*> q;
322 q.push(s);
323
324 while(!q.empty() && !t->isVisited()) {
325 auto v = q.front();
326 q.pop();
327 for(auto e: v->getAdj()) {
328 testAndVisit(q, e, e->getDest(), e->getCapacity() - e->getFlow());
329 }
330 for(auto e: v->getIncoming()) {
331 testAndVisit(q, e, e->getOrig(), e->getFlow());
332 }
333 }
334
335 return t->isVisited();

345double Manager::findMinResidualAlongPath(Vertex* s, Vertex* t) {
346 double f = INT_MAX;
347 for (auto v = t; v != s;) {
348 auto e = v->getPath();
349 if (e->getDest() == v) {
350 f = std::min(f, e->getCapacity() - e->getFlow());
351 v = e->getOrig();
352 }
353 else {
354 f = std::min(f, e->getFlow());
355 v = e->getDest();
356 }
357
358 if (v == s) {
359 break;
360 }
361 }
362 return f;
363}
